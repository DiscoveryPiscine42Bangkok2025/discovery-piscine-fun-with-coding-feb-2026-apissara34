echo '<!DOCTYPE html>
<html>
<head>
    <title>It is over 9000</title>
    <style>
        body { display: flex; justify-content: center; align-items: center; height: 100vh; font-family: sans-serif; }
        .calc-container { padding: 20px; border: 1px solid #ccc; border-radius: 8px; }
        input, select { margin: 5px; padding: 10px; }
    </style>
</head>
<body>
<div class="calc-container">
        <input type="text" id="left" placeholder="Left member">
        <select id="operator">
            <option value="+">+</option>
            <option value="-">-</option>
            <option value=""></option>
            <option value="/">/</option>
            <option value="%">%</option>
        </select>
        <input type="text" id="right" placeholder="Right member">
        <input type="submit" value="Try me!" onclick="calculate()">
    </div>
<script>
        // แจ้งเตือนทุก 30 วินาที
        setInterval(() => {
            alert("Please, use me...");
        }, 30000);

        function calculate() {
            const leftStr = document.getElementById("left").value;
            const rightStr = document.getElementById("right").value;
            const op = document.getElementById("operator").value;

            // ตรวจสอบว่าเป็นตัวเลขจำนวนเต็มบวกหรือไม่
            if (!/^\d+$/.test(leftStr)  !/^\d+$/.test(rightStr)) {
                alert("Error :(");
                return;
            }

            const left = parseInt(leftStr);
            const right = parseInt(rightStr);
            let result;

            // ตรวจสอบการหาร/modulo ด้วยศูนย์
            if ((op === "/"  op === "%") && right === 0) {
                alert("It is over 9000!");
                console.log("It is over 9000!");
                return;
            }

            switch(op) {
                case "+": result = left + right; break;
                case "-": result = left - right; break;
                case "": result = left right; break;
                case "/": result = left / right; break;
                case "%": result = left % right; break;
            }

            alert(result);
            console.log(result);
        }
    </script>

</body>
