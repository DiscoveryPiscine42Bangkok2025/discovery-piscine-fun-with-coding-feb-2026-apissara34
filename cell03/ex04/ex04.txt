echo '<!DOCTYPE html>
<html>
<head>
    <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
</head>
<body>
    <button id="new">New</button>
    <div id="ft_list"></div>
    <script>
        const save = () => {
            let arr = [];
            $("#ft_list div").each(function() { arr.push($(this).text()); });
            document.cookie = "todos=" + JSON.stringify(arr) + ";path=/;max-age=31536000";
        };

        const add = (txt, isNew) => {
            let div = $("<div></div>").text(txt).click(function() {
                if (confirm("Remove?")) { $(this).remove(); save(); }
            });
            isNew ? $("#ft_list").prepend(div) : $("#ft_list").append(div);
        };

        $("#new").click(function() {
            let txt = prompt("New task:");
            if (txt && txt.trim()) { add(txt, true); save(); }
        });

        $(document).ready(function() {
            let c = document.cookie.split("; ").find(r => r.startsWith("todos="));
            if (c) JSON.parse(c.split("=")[1]).forEach(t => add(t, false));
        });
    </script>
</body>
</html>' > ~/discovery_piscine/cell03/ex03bis/index.html
